   pragma solidity ^0.5.1;
   contract machinechange{
    enum State {Ready,Waiting,Active}
    State public state;
    constructor() public {
        state=State.Ready;
    }
    function activate() public {
        state=State.Active;
    }
    function isActive() public view returns(bool){
        return state==State.Active;
    }
   }
